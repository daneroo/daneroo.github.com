--- name: ubuntu-debian-package-management layout: post title: Ubuntu - Debian Package Management time: 2008-04-23 12:26:00.003000 -04:00 tags: [] --- rpm - dpkg equivalents

rpm -qa becomes dpkg -l
rpm -qf becomes dpkg -S
rpm -qal becomes for i in `dpkg-query -W|awk '{print $1}'`; do dpkg -L $i; done
